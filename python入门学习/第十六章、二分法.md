```python
# 算法：最高效解决问题的方法
# 算法二分法
# 需求： 有一个按照从小到大顺序排列的数字列表
# 需要从该数字列表中找到想要的哪一个数字
# 如何做更高效？

# 方案一：整体遍历，效率太低
nums=[-3,4,6,7,20,35,66,88,98]
find_num=20
for i in nums:
    if i == find_num:
        print('找到了')
        break


# 方案二：二分法
# 代码逻辑
def binary_search(find_num,列表):
    mid_val=找到列表中间的值
    if find_num > mid_val:
        # 接下来的查找应该在列表的右半部分
        # 列表=列表切片右半部分
        binary_search(find_num,列表)        # 本身的代码(列表)
    elif find_num < mid_val:
        # 接下来的查找应该在列表的左半部分
        # 列表=列表切片右半部分
        binary_search(find_num,列表)        # 本身的代码(列表)
    else:
        print('找到了')


nums=[-3,4,6,7,10,20,35,66,88,98,100]
find_num=120
def binary_search(find_num,l):
    print(l)  # 打印每一次调用列表的结果
    if len(l) == 0:
        print('找的值不存在')
        return
    mid_index=len(l) // 2
    mid_val=l[mid_index]
    if find_num > mid_val:
        # 接下来的查找应该在列表的右半部分
        l=l[mid_index+1:]
        binary_search(find_num,l)
    elif find_num < mid_val:
        # 接下来的查找应该在列表的左半部分
        l=l[:mid_index]
        binary_search(find_num,l)
    else:
        print('找到了')

binary_search(find_num,nums)

```
